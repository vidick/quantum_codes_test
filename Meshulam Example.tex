\title{Meshulam's example}

\author[1]{Michael Chapman}
\author[2]{Thomas Vidick}
\author[3]{Henry Yuen}
\affil[1]{}
\affil[2]{}
\affil[3]{}

\date{\today}
\maketitle


Let $S=\{x_1,...,x_k\}$ be a set of formal generators. Let $x_i\mapsto A_i$ be the mapping of the generators to permutation matrices acting on $\binom{[k]}{2}\times \field^{k+1}$ as follows. For every pair $j,j'$ where $j<j'$,  and for every $i\neq j$ the matrix $A_i$  acts on $(\{j,j'\},b_1,...,b_{k+1})$ by flipping $b_i$. The matrix $A_j$ flips $b_j$, and conditionally flips $b_{k+1}$ given that $b_{j'}=1$.

Let $B_1,...,B_k$ be a collection of commuting order $2$ unitaries in $U(d')$, where $d'\geq d:=2^{k+1}\binom{k}{2}$. Assume further that $\complex^d$ was embedded in $\complex^{d'}$ and that $P_d$ is the (orthogonal) projection to the copy of $\complex^d$ in $\complex^{d'}$. Let $\Vert A\Vert^2_F=Tr(A^*A)$ be the Frobenius norm of $A$ and $\Vert A\Vert^2_{hs}=\frac{1}{\dim A}\Vert A\Vert^2_F$ be the Hilbert--Schmidt norm.\footnote{It is common to call $\Vert \cdot \Vert_F$ either the Frobenius or the Hilbert--Schmidt norm, and $\Vert \cdot \Vert_{hs}$ their normalized versions. The reason we call the normalized version Hilbert--Schmidt and the un-normalized Frobenius is because it became common in stability theory --- see Lubotzky.....} Furthermore, let $\mB_{j,j'}$ be the strings with fixed pair $\{j,j'\}$ as prefix, $V_{j,j'}$ the linear subspace spanned by $\mB_{j,j'}$ in $\complex^d$, and $P_{j,j'}$ the projection onto this subspace. Note that 
\[
\Vert A\Vert_{hs}^2=\Expectation_{e_i} [\Vert Ae_i\Vert_2^2]\ ,\quad \Vert A\Vert_{F}^2=\sum_{e_i}\Vert Ae_i\Vert_2^2.
\]
Now, for every  $j<j'$ and $i<i'$, if $(i,i')\neq (j,j')$, then the restriction of $(A_jA_{j'})^2$ to $V_{i,i'}$ is the identity. On the other hand, the squared Hilbert--Schmidt distance between $(A_jA_{j'})^2$ and $\Id_{j,j'}$ is $2$. Therefore, for every  $j<j'$,
\[
\begin{split}
  \Vert A_jA_{j'}A_jA_{j'}-\Id_{d}\Vert_{hs}^2  =\Expectation_{i<i'}[\Vert P_{i,i'}A_jA_{j'}A_jA_{j'}P_{i,i'}-\Id_{i,i'}\Vert_{hs}^2 ]=\frac{1}{\binom{k}{2}}\Vert P_{j,j'}A_jA_{j'}A_jA_{j'}P_{j,j'}-\Id_{j,j'}\Vert_{hs}^2=\nicefrac{2}{\binom{k}{2}}.
\end{split}
\]
On the other hand, for every orthogonal projection $P$ on $\textrm{Span}\{e_1,...,e_t\}$, such that the $PA_iP$'s act as permutations on $\{e_1,...,e_t\}$ (and in particular $A_i$ commutes with $P$),  we have
\[
\begin{split}
    \Vert P(A_j-B_j)\Vert_F^2&=\Vert P(A_j-B_j)P\Vert_F^2+\Vert P(A_j-B_j)(I-P)\Vert_F^2\\
    &=\Vert P(\Id-A_jB_j)P\Vert_F^2+\Vert PB_j(I-P)\Vert_F^2.
\end{split}
\]
Now, for every $1\leq i\leq t$, 
\[
1-|e_i^*A_jB_je_i|^2\leq |1-e_i^*A_jB_je_i|^2\leq \Vert e_i^*(I-A_jB_j)P\Vert_2^2,
\]
and
\[
\begin{split}
 1-|e_i^*A_jB_je_i|^2&=\sum_{r\neq i}|e_i^*A_jB_je_r|^2\\   
 &\geq \Vert e_i^*A_jB_j(\Id-P)\Vert_2^2.
\end{split}
\]
Since $A_j$ permutes $\{e_1,...,e_t\}$, we have
\[
\begin{split}
 \Vert PB_j(I-P)\Vert_F^2&=   \Vert PA_jB_j(I-P)\Vert_F^2\\
 &=\sum_{i=1}^t \Vert e_i^*A_jB_j(\Id-P)\Vert_2^2\\
 &\leq \sum_{i=1}^t\Vert e_i^*(I-A_jB_j)P\Vert_2^2\\
 &=\Vert P(I-A_jB_j)P\Vert_F^2.
\end{split}
\]
All in all, 
\[
 \Vert P(A_j-B_j)\Vert_F^2\leq 2 \Vert P(A_j-B_j)P\Vert_F^2.
\]
Thus,
\begin{align*}
    \Vert PA_jA_{j'}A_jA_{j'}P-P\Vert_F&=\Vert PA_jA_{j'}A_jA_{j'}P-PB_jB_{j'}B_jB_{j'}P\Vert_F\\
    &\leq \Vert P(A_j-B_j)B_{j'}B_jB_{j'}P\Vert_F+\Vert PA_j(A_{j'}-B_{j'})B_jB_{j'}P\Vert_F\\
    &+\Vert PA_jA_{j'}(A_{j}-B_{j})B_{j'}P\Vert_F+\Vert PA_jA_{j'}A_j(A_{j'}-B_{j'})P\Vert_F\\
    &= 2\Vert P(A_j-B_j)\Vert_F+2\Vert P(A_{j'}-B_{j'})\Vert_F\\
    &\leq 2\sqrt{2}\Vert P(A_j-B_j)P\Vert_F+2\sqrt{2}\Vert P(A_{j'}-B_{j'})P\Vert_F
\end{align*}
Combining what we have, we get
\[
\begin{split}
  \nicefrac{2}{\binom{k}{2}}&=\max_{j<j'} \Vert A_jA_{j'}A_jA_{j'}-\Id_d\Vert_{hs}^2\\
  &= \frac{1}{d^2}\Expectation_{j<j'}\Vert A_jA_{j'}A_jA_{j'}-\Id_d\Vert_F^2\\
  &=\frac{1}{d^2}\Expectation_{j<j'}\Vert P_{j,j'}A_jA_{j'}A_jA_{j'}P_{j,j'}-P_{j,j'}\Vert_F^2\\
  &\leq\frac{1}{d^2}\Expectation_{j<j'}(2\sqrt{2}\Vert P_{j,j'}(A_j-B_j)P_{j,j'}\Vert_F+2\sqrt{2}\Vert P_{j,j'}(A_{j'}-B_{j'})P_{j,j'}\Vert_F)^2\\
  &\leq \frac{1}{d^2}\Expectation_{j<j'}16\Vert P_{j,j'}(A_j-B_j)P_{j,j'}\Vert^2_F+16\Vert P_{j,j'}(A_{j'}-B_{j'})P_{j,j'}\Vert_F^2\\
  &=\frac{16}{d^2\binom{k}{2}}\sum_{j}\underbrace{\sum_{j'\neq j}\Vert P_{j,j'}(A_j-B_j)P_{j,j'}\Vert^2_F}_{\leq \Vert P_d(A_j-B_j)P_d\Vert_F^2}\\
  &\leq \frac{16k}{\binom{k}{2}}\Expectation_j\Vert P_d(A_j-B_j)P_d\Vert_F^2.
\end{split}
\]
This means that 
\[
\Expectation_j\Vert P_d(A_j-B_j)P_d\Vert_F^2\geq \frac{1}{8k},
\]
and we are done.
